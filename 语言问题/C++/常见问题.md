# 读取文件操作

```C++
//读取文件的操作
void getTxt(string str, vector<pair<int, int>>& need){
    //实例化一个文件类
    fstream Myfile;
    //以只读的方式打开文件
    Myfile.open(str, ios::in);

    //读取的数字放到vector数组中
    vector<int> getFile;
    //如果文件打开成功了
    if(Myfile.is_open()){
        //定义一个变量，每行的字符串放到变量中
        string buffer;
        while(getline(Myfile,buffer)){
            getFile.push_back(stoi(buffer));
        }  
    }
    Myfile.close();

    //获得需要查询的节点对
    int m = 0, n = 1;
    while(getFile[m] != -1){
        need.push_back(make_pair(getFile[m], getFile[n]));
        m += 2;
        n += 2;
    }
    return;
}
```
